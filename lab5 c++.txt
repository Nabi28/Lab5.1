#include <isostream>	
using namespace std;
int main()
{
	int a, b, c;
	cout << "Enter three integers: ";
	cin >> a >> b >> c;
	if (a > b && a > c)
		cout << "The largest number is: " << a << endl;
	else if (b > a && b > c)
		cout << "The largest number is: " << b << endl;
	else if (c > a && c > b)
		cout << "The largest number is: " << c << endl;
	else
		cout << "There is no single largest number." << endl;
	return 0;
}
